162 /*
163  * Interface infinite loop
164  * monitor tun interface and reverse deque by means of selec syscall
165  */
166 void Iface::Run()
167 {
168  logh->Log("[Iface::Run]");
169  timeval ts;
170  fd_set rds, wds;
171  int rd=dh->FdReverse();
172  while (!Check::DConf::GetReset()) {
173  ts = {confh->Guard()*AF/F, U};
174  auto maxfd=std::max(id, rd)+1;
175  FD_ZERO(&rds);
176  FD_ZERO(&wds);
177  FD_SET(id, &rds);
178  FD_SET(rd, &rds);
179  select(maxfd, &rds, 0, 0, &ts);
180  if (FD_ISSET(id, &rds) && !Check::DConf::GetReset())
181  Read();
182  if (FD_ISSET(rd, &rds) && !Check::DConf::GetReset())
183  Write();
184  }
185 }
