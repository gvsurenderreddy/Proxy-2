56 int ProtoBase::Lenght(char *_buff) const
57 {

73 /*
74  * Recv I/O
75  * receiving in the outer socket writes in the reverse deque
76  */
77 void ProtoBase::Recv(std::shared_ptr<Layer::BaseSock> _fd, int *_flag)
78 {
79  char *buff=nullptr;
80  int n=mp->Alloc(buff);
81  if (!n) {
82  logh->Exc("[ProtoBase::Recv]: no memory blocks available");
83  return;
84  }
85  _fd->SetBuff(buff);
86  _fd->SetN(n);
87  int nread;
88  if ((nread=sh->Recv(_fd))>0 && !_fd->GetError()->Get()) {
89  logh->Log("[ProtoBase::Recv]:", nread);
90  logh->Hex(buff, nread);
91  int plen=Lenght(buff);
92  if (plen != nread)
93  logh->Exc("[ProtoBase::Lenght]:", nread, "!=", plen);
94  else
95  logh->Log("[ProtoBase::Lenght]: bce xopowo");
96  dh->PushReverse(nread, buff);
97  } else {
98  logh->Log("[ProtoBase::Recv]: EOF", *_flag);
99  mp->Dealloc(buff);
100  *_flag = 0;
101  }
102 }

104 /*
105  * Send I/O
106  * receiving in the direct deque writes in the outer socket
107  */
108 void ProtoBase::Send(std::shared_ptr<Layer::BaseSock> _fd, int *_flag)
109 {
110  logh->Log("[ProtoBase::Send]: send", *_flag);
111  auto de=dh->FrontDirect();
112  if (de) {
113  _fd->SetBuff(de->second);

116  logh->Exc("[ProtoBase::Send]: EOF", *_flag);

118  }
119  mp->Dealloc(de->second);
120  dh->PopDirect();
121  }
122  dh->CheckDirect();
123 }

203 /*
204  * Client Selector:
205  * monitor status of direct deque and socket file descriptors
206  */
207 void StreamClient::Selector()
208 {
209  timeval ts;
210  fd_set rds, wds;
211  int dd=dh->FdDirect();
212  int fd=sd->GetSd(0);
213  int tfd=fd;
214  logh->Log("[StreamServer::Selector]:", fd);
215  while (fd && !Check::DConf::GetReset() && !sd->GetError()->Get()) {
216  ts = {confh->Guard()*AF/F, U};
217  auto maxfd=std::max(dd, fd)+1;
218  FD_ZERO(&rds);
219  FD_ZERO(&wds);
220  FD_SET(fd, &rds);
221  FD_SET(dd, &rds);
222  select(maxfd, &rds, 0, 0, &ts);
223  if (FD_ISSET(fd, &rds) && !Check::DConf::GetReset() &&
224  !sd->GetError()->Get())
225  Recv(sd, &fd);
226  if (FD_ISSET(dd, &rds) && !Check::DConf::GetReset() &&
227  !sd->GetError()->Get())
228  Send(sd, &fd);
229  }
230  Check::CConf::Status(false);
231  logh->Log("[StreamServer::Selector]: removing", tfd);
232  if (!((confh->Key()).empty() || (confh->Cert()).empty()))
233  sh->Close(tfd, sd->GetSsl());
234  else
235  sh->Close(tfd);
236 }

343 void StreamServer::HandleConn()
344 {

348  auto st=std::make_shared<Threads::SmartThread>([](void *_t)->void* {

361  sh->Close(sd->GetSd(0));
362  st->Join();
363  Init();

366 /*
367  * Server Selector:
368  * monitor status of direct deque and socket file descriptors
369  * as server is multi-thread, each thread is responsible of its own socket (csd)
370  */
371 void StreamServer::Selector()
372 {
373  auto tt=tp->Get(Threads::ThreadBase::Pself());
374  logh->Log("[StreamServer::Selector]: getting for", tt->Uid());
375  int fd=csd[tt->Uid()];
376  int dd=dh->FdDirect();
377  int tfd=fd;
378  auto tsd=sp->Get(fd);
379  logh->Log("[StreamServer::Selector]:", fd);
380  timeval ts;
381  fd_set rds, wds;
382  while (fd && !Check::DConf::GetReset() && !tsd->GetError()->Get()) {
383  ts = {confh->Guard()*AF/F, U};
384  auto maxfd=std::max(dd, fd)+1;
385  FD_ZERO(&rds);
386  FD_ZERO(&wds);
387  FD_SET(fd, &rds);
388  FD_SET(dd, &rds);
389  select(maxfd, &rds, 0, 0, &ts);
390  if (FD_ISSET(fd, &rds) && !Check::DConf::GetReset() &&
391  !tsd->GetError()->Get())
392  Recv(tsd, &fd);
393  if (FD_ISSET(dd, &rds) && !Check::DConf::GetReset() &&
394  !sd->GetError()->Get())
395  Send(tsd, &fd);

466 /*
467  * Accept connections from clients and handle them
468  */
469 void StreamServer::Run()
470 {
471  logh->Log("[StreamServer::Run]");
472  sp->Set(sd->GetSd(0), sd);
473  while (!Check::DConf::GetReset() && !sd->GetError()->Get()) {
474  timeval ts={confh->Guard()*AF/F, U};
475  fd_set rds;
476  auto maxfd=sd->GetSd(0)+1;
477  FD_ZERO(&rds);
478  FD_SET(sd->GetSd(0), &rds);
479  select(maxfd, &rds, 0, 0, &ts);
480  if (!Check::DConf::GetReset() &&
481  FD_ISSET(sd->GetSd(0), &rds)) {
482  int err=exc->run([this]()->int {
483  if (Check::DConf::GetReset())
484  return -1;
485  this->sh->Accept(this->sd);
486  if (this->sd->GetError()->Get() ||
487  (this->sd->GetSd(1) <= 0))
488  return -1;
489  Check::CConf::Status(true);
490  return 0;
491  });
492  if (!err)
493  HandleConn();
494  else
495  logh->Exc("[StreamServer::Run]:",
496  "error accepting new connection");
497  }
498  Summary();
499  }
500  if (!Check::DConf::GetReset())
501  Check::DConf::SetReset();
