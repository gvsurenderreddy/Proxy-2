144 /*

147 void Nix::Tloop()
148 {
149  logh->Log("[Nix::Tloop]: preparing threads");
150  timeval tp={confh->Guard()*TF/F, U};
151  while (!Check::DConf::GetConfig() && !Check::DConf::GetReset()) {
152  timeval tv=tp;
153  if (iflag)
154  Cinner();
155  if (oflag)
156  Couter();
157  select(0, 0, 0, 0, &tv);
158  Cycle();
159  Summary();
160  }
161  Clean();
162 }
