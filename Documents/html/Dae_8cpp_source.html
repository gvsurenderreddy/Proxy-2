185 /*
186  * Outer socket initialization:
187  * configure the Chain of Responsibility (TCP/IP stack) and its message (sd)
188  * IPv6 server on dual-stack host can service both IPv6 and IPv4
189  * IPv6 client on dual-stack host can communicate with IPv6 and IPv4 servers
190  * ... hence -> IPv6
191  */
192 void OuterSocket::Init()
193 {
194  logh->Log("[OuterSocket::Init]");
195  if (confh->Client())
196  app = std::make_shared<StreamClient>();
197  else
198  app = std::make_shared<StreamServer>();
199  tls = std::make_shared<Layer::SockTls>();
200  tcp = std::make_shared<Layer::SockTcp>();
201  ip = std::make_shared<Layer::SockIp<Layer::IA6,Layer::SAI6>>(IPv6);
202  sd = std::make_shared<Layer::ActiveSock<Layer::IA6,Layer::SAI6>>(IPv6);
203  tls->SetSucc(tcp);
204  tcp->SetSucc(ip);
205  if (confh->Client())
206  InitClient();
207  else
208  InitServer();
209  InitSsl();
210  app->SetHandler(tls);
211  app->SetSocket(sd);
212  app->Init();
213 }
